def add(name, value)
  name = name.to_sym
  raise "string #{name} is already set to #{value}" if strings.has_key?(name) && self[name] != value
  replace name, value
end

def inject_to(obj)
  @objects_to_update << obj
  strings.keys.each { |name| update_object obj, name }
end

def replace(name, value)
  name          = name.to_sym
  strings[name] = value
  update_objects name
end
