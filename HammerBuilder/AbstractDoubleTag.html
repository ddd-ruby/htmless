def with
  flush_classes
  @output << GT
  @content         = nil
  @builder.current = nil
  #yield
  if (content = yield).is_a?(String)
    @output << CGI.escapeHTML(content)
  end
  @builder.flush
  @output << SLASH_LT << @stack.pop << GT
  nil
end
